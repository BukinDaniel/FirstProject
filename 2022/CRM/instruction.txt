git status - показать статус измененных объектов
git add . - добавить измененные объекты в пуш
git commit -m 'НазваниеИзменений' - закоммитить изменения
git push - отправить изменения в репозиторий
git checkout имяветки - переключение текущей ветки
git merge dev(надо быть в ветке master) - для переноса изменений с ветки dev в ветку master
